-- ============================================================
--  NHÓM 1: AUTH & CONFIG

CREATE TABLE Admins (
    AdminId         INT             PRIMARY KEY IDENTITY(1,1),
    Username        NVARCHAR(100)   NOT NULL UNIQUE,
    PasswordHash    NVARCHAR(256)   NOT NULL,   -- BCrypt hash
    FullName        NVARCHAR(200)   NULL,
    IsActive        BIT             NOT NULL DEFAULT 1,
    CreatedAt       DATETIME2       NOT NULL DEFAULT GETDATE(),
    LastLoginAt     DATETIME2       NULL
);
GO

CREATE TABLE Tables (
    TableId     INT             PRIMARY KEY IDENTITY(1,1),
    TableCode   NVARCHAR(20)    NOT NULL UNIQUE,    -- 'B01', 'B02', 'TERRACE01'
    TableName   NVARCHAR(100)   NULL,
    Floor       INT             NOT NULL DEFAULT 1,
    Capacity    INT             NOT NULL DEFAULT 4,
    IsActive    BIT             NOT NULL DEFAULT 1,
    CreatedAt   DATETIME2       NOT NULL DEFAULT GETDATE()
);
GO

-- Lưu cấu hình từng thiết bị tablet (thiết lập 1 lần)
CREATE TABLE TabletConfig (
    ConfigId    INT             PRIMARY KEY IDENTITY(1,1),
    DeviceId    NVARCHAR(200)   NOT NULL UNIQUE,   -- Android device ID
    Role        NVARCHAR(20)    NOT NULL            -- 'order' | 'kitchen'
                CONSTRAINT CK_TabletConfig_Role CHECK (Role IN ('order','kitchen')),
    TableId     INT             NULL REFERENCES Tables(TableId),  -- NULL nếu là bếp
    SetupAt     DATETIME2       NOT NULL DEFAULT GETDATE(),
    UpdatedAt   DATETIME2       NULL
);
GO

-- ============================================================
--  NHÓM 2: MENU & NGUYÊN LIỆU
-- ============================================================

CREATE TABLE Categories (
    CategoryId  INT             PRIMARY KEY IDENTITY(1,1),
    Name        NVARCHAR(100)   NOT NULL,
    IconUrl     NVARCHAR(500)   NULL,
    SortOrder   INT             NOT NULL DEFAULT 0,
    IsActive    BIT             NOT NULL DEFAULT 1
);
GO

CREATE TABLE MenuItems (
    MenuItemId      INT             PRIMARY KEY IDENTITY(1,1),
    CategoryId      INT             NOT NULL REFERENCES Categories(CategoryId),
    Name            NVARCHAR(200)   NOT NULL,
    Description     NVARCHAR(500)   NULL,
    Price           DECIMAL(10,2)   NOT NULL
                    CONSTRAINT CK_MenuItems_Price CHECK (Price >= 0),
    ImageUrl        NVARCHAR(500)   NULL,
    IsAvailable     BIT             NOT NULL DEFAULT 1,
    StockQty        INT             NOT NULL DEFAULT 0,     -- số lượng có thể phục vụ hôm nay
    SortOrder       INT             NOT NULL DEFAULT 0,
    CreatedAt       DATETIME2       NOT NULL DEFAULT GETDATE(),
    UpdatedAt       DATETIME2       NULL
);
GO

CREATE TABLE Ingredients (
    IngredientId    INT             PRIMARY KEY IDENTITY(1,1),
    Name            NVARCHAR(200)   NOT NULL,
    Unit            NVARCHAR(10)    NOT NULL
                    CONSTRAINT CK_Ingredients_Unit CHECK (Unit IN ('gram','ml')),
    CurrentStock    DECIMAL(10,2)   NOT NULL DEFAULT 0,
    MinStock        DECIMAL(10,2)   NOT NULL DEFAULT 0,     -- ngưỡng cảnh báo sắp hết
    CreatedAt       DATETIME2       NOT NULL DEFAULT GETDATE(),
    UpdatedAt       DATETIME2       NULL
);
GO

-- Công thức: 1 món cần bao nhiêu nguyên liệu
CREATE TABLE ItemIngredients (
    ItemIngredientId    INT             PRIMARY KEY IDENTITY(1,1),
    MenuItemId          INT             NOT NULL REFERENCES MenuItems(MenuItemId),
    IngredientId        INT             NOT NULL REFERENCES Ingredients(IngredientId),
    AmountPerServing    DECIMAL(10,2)   NOT NULL
                        CONSTRAINT CK_ItemIngredients_Amount CHECK (AmountPerServing > 0),
    CONSTRAINT UQ_ItemIngredients UNIQUE (MenuItemId, IngredientId)
);
GO

-- Lịch sử nhập / xuất kho
CREATE TABLE StockTransactions (
    TransactionId   INT             PRIMARY KEY IDENTITY(1,1),
    IngredientId    INT             NOT NULL REFERENCES Ingredients(IngredientId),
    TransType       NVARCHAR(10)    NOT NULL
                    CONSTRAINT CK_StockTrans_Type CHECK (TransType IN ('in','out')),
    Amount          DECIMAL(10,2)   NOT NULL
                    CONSTRAINT CK_StockTrans_Amount CHECK (Amount > 0),
    Reason          NVARCHAR(300)   NULL,    -- 'Nhập kho sáng' | 'Order #123' | 'Hao hụt'
    OrderItemId     INT             NULL,    -- nếu xuất kho do order
    CreatedAt       DATETIME2       NOT NULL DEFAULT GETDATE()
);
GO

--  NHÓM 3: ORDER & THANH TOÁN

-- Phiên ngồi (khách vào = mở session, khách về = đóng session)
CREATE TABLE TableSessions (
    SessionId       INT             PRIMARY KEY IDENTITY(1,1),
    TableId         INT             NOT NULL REFERENCES Tables(TableId),
    Status          NVARCHAR(20)    NOT NULL DEFAULT 'open'
                    CONSTRAINT CK_Sessions_Status CHECK (Status IN ('open','closed')),
    OpenedAt        DATETIME2       NOT NULL DEFAULT GETDATE(),
    ClosedAt        DATETIME2       NULL
);
GO

-- Đơn hàng (CHỈ INSERT khi ZaloPay xác nhận thành công)
CREATE TABLE Orders (
    OrderId             INT             PRIMARY KEY IDENTITY(1,1),
    SessionId           INT             NOT NULL REFERENCES TableSessions(SessionId),
    TableId             INT             NOT NULL REFERENCES Tables(TableId),
    TableCode           NVARCHAR(20)    NOT NULL,
    TotalAmount         DECIMAL(12,2)   NOT NULL,
    Status              NVARCHAR(20)    NOT NULL DEFAULT 'preparing'
                        CONSTRAINT CK_Orders_Status CHECK (Status IN ('preparing','done')),
    ZaloPayOrderId      NVARCHAR(200)   NULL UNIQUE,   -- app_trans_id từ ZaloPay
    Note                NVARCHAR(500)   NULL,
    CreatedAt           DATETIME2       NOT NULL DEFAULT GETDATE(),
    CompletedAt         DATETIME2       NULL
);
GO

CREATE TABLE OrderItems (
    OrderItemId     INT             PRIMARY KEY IDENTITY(1,1),
    OrderId         INT             NOT NULL REFERENCES Orders(OrderId),
    MenuItemId      INT             NOT NULL REFERENCES MenuItems(MenuItemId),
    ItemName        NVARCHAR(200)   NOT NULL,   -- snapshot tên lúc order
    UnitPrice       DECIMAL(10,2)   NOT NULL,   -- snapshot giá lúc order
    Quantity        INT             NOT NULL    DEFAULT 1
                    CONSTRAINT CK_OrderItems_Qty CHECK (Quantity > 0),
    Note            NVARCHAR(300)   NULL,       -- 'ít đá', 'nhiều đường', 'không đường'
    ItemStatus      NVARCHAR(20)    NOT NULL    DEFAULT 'pending'
                    CONSTRAINT CK_OrderItems_Status CHECK (ItemStatus IN ('pending','done'))
);
GO

-- Thanh toán ZaloPay
CREATE TABLE Payments (
    PaymentId           INT             PRIMARY KEY IDENTITY(1,1),
    OrderId             INT             NOT NULL REFERENCES Orders(OrderId),
    ZaloPayOrderId      NVARCHAR(200)   NULL,
    ZaloTransId         NVARCHAR(200)   NULL,   -- zp_trans_id từ webhook
    Amount              DECIMAL(12,2)   NOT NULL,
    Method              NVARCHAR(20)    NOT NULL DEFAULT 'zalopay',
    Status              NVARCHAR(20)    NOT NULL DEFAULT 'success'
                        CONSTRAINT CK_Payments_Status CHECK (Status IN ('pending','success','failed')),
    PaidAt              DATETIME2       NOT NULL DEFAULT GETDATE(),
    WebhookPayload      NVARCHAR(MAX)   NULL    -- raw JSON từ ZaloPay để audit/đối soát
);
GO

-- Hàng đợi bếp (real-time, polling mỗi 3s)
CREATE TABLE KitchenQueue (
    QueueId         INT             PRIMARY KEY IDENTITY(1,1),
    OrderId         INT             NOT NULL REFERENCES Orders(OrderId),
    OrderItemId     INT             NOT NULL REFERENCES OrderItems(OrderItemId),

    ItemName        NVARCHAR(200)   NOT NULL,
    Quantity        INT             NOT NULL,
    Note            NVARCHAR(300)   NULL,
    Status          NVARCHAR(20)    NOT NULL DEFAULT 'waiting'
                    CONSTRAINT CK_KitchenQueue_Status CHECK (Status IN ('waiting','done')),
    OrderedAt       DATETIME2       NOT NULL DEFAULT GETDATE(),
    DoneAt          DATETIME2       NULL
);
GO

-- Đăng nhập admin
CREATE OR ALTER PROCEDURE sp_Admin_Login
    @Username   NVARCHAR(100),
    @ReturnHash NVARCHAR(256) OUTPUT  -- trả hash về .NET để BCrypt.Verify
AS
BEGIN
    SET NOCOUNT ON;
    SELECT @ReturnHash = PasswordHash
    FROM Admins
    WHERE Username = @Username AND IsActive = 1;

    IF @ReturnHash IS NULL
        RETURN 0;   -- không tìm thấy

    UPDATE Admins SET LastLoginAt = GETDATE() WHERE Username = @Username;
    RETURN 1;
END
GO

-- ============================================================
--  SP NHÓM: TABLET CONFIG
-- ============================================================

CREATE OR ALTER PROCEDURE sp_TabletConfig_Setup
    @DeviceId   NVARCHAR(200),
    @Role       NVARCHAR(20),
    @TableCode  NVARCHAR(20) = NULL   -- chỉ cần nếu Role='order'
AS
BEGIN
    SET NOCOUNT ON;
    DECLARE @TableId INT = NULL;

    IF @Role = 'order'
    BEGIN
        SELECT @TableId = TableId FROM Tables WHERE TableCode = @TableCode AND IsActive = 1;
        IF @TableId IS NULL
        BEGIN
            RAISERROR('Mã bàn không tồn tại: %s', 16, 1, @TableCode);
            RETURN;
        END
    END

    MERGE TabletConfig AS target
    USING (SELECT @DeviceId AS DeviceId) AS src ON target.DeviceId = src.DeviceId
    WHEN MATCHED THEN
        UPDATE SET Role = @Role, TableId = @TableId, UpdatedAt = GETDATE()
    WHEN NOT MATCHED THEN
        INSERT (DeviceId, Role, TableId) VALUES (@DeviceId, @Role, @TableId);
END
GO

CREATE OR ALTER PROCEDURE sp_TabletConfig_Get
    @DeviceId NVARCHAR(200)
AS
BEGIN
    SET NOCOUNT ON;
    SELECT tc.ConfigId, tc.DeviceId, tc.Role, tc.TableId,
           t.TableCode, t.TableName, tc.SetupAt
    FROM TabletConfig tc
    LEFT JOIN Tables t ON t.TableId = tc.TableId
    WHERE tc.DeviceId = @DeviceId;
END
GO

-- Mở session khi khách vào (hoặc lấy session đang open)
CREATE OR ALTER PROCEDURE sp_Session_OpenOrGet
    @TableCode  NVARCHAR(20),
    @SessionId  INT OUTPUT
AS
BEGIN
    SET NOCOUNT ON;
    DECLARE @TableId INT;
    SELECT @TableId = TableId FROM Tables WHERE TableCode = @TableCode AND IsActive = 1;
    IF @TableId IS NULL
    BEGIN
        RAISERROR('Bàn không tồn tại', 16, 1);
        RETURN;
    END

    -- Nếu đang có session open thì dùng lại
    SELECT @SessionId = SessionId
    FROM TableSessions
    WHERE TableId = @TableId AND Status = 'open';

    IF @SessionId IS NULL
    BEGIN
        INSERT INTO TableSessions (TableId) VALUES (@TableId);
        SET @SessionId = SCOPE_IDENTITY();
    END
END
GO

CREATE OR ALTER PROCEDURE sp_Menu_GetAll
    @IncludeUnavailable BIT = 0
AS
BEGIN
    SET NOCOUNT ON;
    SELECT mi.MenuItemId, mi.CategoryId, c.Name AS CategoryName,
           mi.Name, mi.Description, mi.Price, mi.ImageUrl,
           mi.IsAvailable, mi.StockQty, mi.SortOrder
    FROM MenuItems mi
    INNER JOIN Categories c ON c.CategoryId = mi.CategoryId
    WHERE c.IsActive = 1
      AND (@IncludeUnavailable = 1 OR mi.IsAvailable = 1)
    ORDER BY c.SortOrder, mi.SortOrder, mi.Name;
END
GO

CREATE OR ALTER PROCEDURE sp_Menu_GetById
    @MenuItemId INT
AS
BEGIN
    SET NOCOUNT ON;
    SELECT mi.MenuItemId, mi.CategoryId, c.Name AS CategoryName,
           mi.Name, mi.Description, mi.Price, mi.ImageUrl,
           mi.IsAvailable, mi.StockQty, mi.SortOrder,
           mi.CreatedAt, mi.UpdatedAt
    FROM MenuItems mi
    INNER JOIN Categories c ON c.CategoryId = mi.CategoryId
    WHERE mi.MenuItemId = @MenuItemId;

    -- Kèm nguyên liệu
    SELECT ii.ItemIngredientId, ii.IngredientId, i.Name AS IngredientName,
           ii.AmountPerServing, i.Unit
    FROM ItemIngredients ii
    INNER JOIN Ingredients i ON i.IngredientId = ii.IngredientId
    WHERE ii.MenuItemId = @MenuItemId;
END
GO

CREATE OR ALTER PROCEDURE sp_Menu_Update
    @MenuItemId     INT,
    @CategoryId     INT,
    @Name           NVARCHAR(200),
    @Description    NVARCHAR(500) = NULL,
    @Price          DECIMAL(10,2),
    @ImageUrl       NVARCHAR(500) = NULL,
    @IsAvailable    BIT,
    @StockQty       INT,
    @SortOrder      INT = 0
AS
BEGIN
    SET NOCOUNT ON;

        StockQty     = @StockQty,
        SortOrder    = @SortOrder,

        RAISERROR('Không tìm thấy món: %d', 16, 1, @MenuItemId);

GO

CREATE OR ALTER PROCEDURE sp_Menu_Delete
    @MenuItemId INT
AS
BEGIN
    SET NOCOUNT ON;
    -- Soft delete: chỉ ẩn, không xoá vì đã có trong OrderItems
    UPDATE MenuItems SET IsAvailable = 0, UpdatedAt = GETDATE()
    WHERE MenuItemId = @MenuItemId;
END
GO

CREATE OR ALTER PROCEDURE sp_Menu_UpdateStock
    @MenuItemId INT,
    @StockQty   INT
AS
BEGIN
    SET NOCOUNT ON;
    UPDATE MenuItems
    SET StockQty    = @StockQty,
        IsAvailable = CASE WHEN @StockQty > 0 THEN 1 ELSE 0 END,
        UpdatedAt   = GETDATE()
    WHERE MenuItemId = @MenuItemId;
END
GO

AS

CREATE OR ALTER PROCEDURE sp_Ingredient_AdjustStock
    @IngredientId   INT,
    @TransType      NVARCHAR(10),   -- 'in' | 'out'
    @Amount         DECIMAL(10,2),
    @Reason         NVARCHAR(300) = NULL,
    @OrderItemId    INT = NULL
AS
BEGIN
    SET NOCOUNT ON;
    BEGIN TRANSACTION;
    BEGIN TRY
        IF @TransType = 'in'
            UPDATE Ingredients
            SET CurrentStock = CurrentStock + @Amount, UpdatedAt = GETDATE()
            WHERE IngredientId = @IngredientId;
        ELSE
        BEGIN
            DECLARE @cur DECIMAL(10,2);
            SELECT @cur = CurrentStock FROM Ingredients WHERE IngredientId = @IngredientId;

            IF @cur < @Amount
            BEGIN
                ROLLBACK;
                -- Dùng THROW với message tự build, tránh lỗi RAISERROR với DECIMAL
                DECLARE @msg NVARCHAR(500) = CONCAT(
                    N'Không đủ nguyên liệu (hiện có: ',
                    CAST(@cur AS NVARCHAR(20)),
                    N', cần: ',
                    CAST(@Amount AS NVARCHAR(20)),
                    N')'
                );
                THROW 50001, @msg, 1;
                RETURN;
            END

            UPDATE Ingredients
            SET CurrentStock = CurrentStock - @Amount, UpdatedAt = GETDATE()
            WHERE IngredientId = @IngredientId;
        END

        INSERT INTO StockTransactions (IngredientId, TransType, Amount, Reason, OrderItemId)
        VALUES (@IngredientId, @TransType, @Amount, @Reason, @OrderItemId);

        COMMIT;
    END TRY
    BEGIN CATCH
        IF XACT_STATE() <> 0
            ROLLBACK;
        THROW;
    END CATCH
END
GO

-- Tạo pending order trước khi thanh toán (để lưu ZaloPayOrderId)
-- Trả về OrderId để ghép với ZaloPay app_trans_id
CREATE OR ALTER PROCEDURE sp_Order_CreatePending
    @TableCode      NVARCHAR(20),
    @ZaloPayOrderId NVARCHAR(200),
    @TotalAmount    DECIMAL(12,2),
    @Note           NVARCHAR(500) = NULL,
    @OrderId        INT OUTPUT
AS
BEGIN
    SET NOCOUNT ON;
    DECLARE @TableId INT, @SessionId INT;

    SELECT @TableId = TableId FROM Tables WHERE TableCode = @TableCode AND IsActive = 1;
    IF @TableId IS NULL
    BEGIN
        RAISERROR('Bàn không tồn tại', 16, 1); RETURN;
    END

    EXEC sp_Session_OpenOrGet @TableCode, @SessionId OUTPUT;

    INSERT INTO Orders (SessionId, TableId, TableCode, TotalAmount, Status, ZaloPayOrderId, Note)
    VALUES (@SessionId, @TableId, @TableCode, @TotalAmount, 'preparing', @ZaloPayOrderId, @Note);

    SET @OrderId = SCOPE_IDENTITY();
END
GO

-- Được gọi từ ZaloPay Webhook sau khi thanh toán thành công
CREATE OR ALTER PROCEDURE sp_Order_ConfirmPayment
    @ZaloPayOrderId     NVARCHAR(200),
    @ZaloTransId        NVARCHAR(200),
    @PaidAmount         DECIMAL(12,2),
    @WebhookPayload     NVARCHAR(MAX),
    @ItemsJson          NVARCHAR(MAX)   -- JSON: [{menuItemId,name,price,qty,note}]
AS
BEGIN
    SET NOCOUNT ON;
    BEGIN TRANSACTION;
    BEGIN TRY
        DECLARE @OrderId INT;
        SELECT @OrderId = OrderId FROM Orders WHERE ZaloPayOrderId = @ZaloPayOrderId;
        IF @OrderId IS NULL
        BEGIN
            ROLLBACK;
            RAISERROR('Không tìm thấy order với ZaloPayOrderId: %s', 16, 1, @ZaloPayOrderId);
            RETURN;
        END

        -- 1. Ghi payment
        INSERT INTO Payments (OrderId, ZaloPayOrderId, ZaloTransId, Amount, Status, WebhookPayload)
        VALUES (@OrderId, @ZaloPayOrderId, @ZaloTransId, @PaidAmount, 'success', @WebhookPayload);

        -- 2. Insert OrderItems từ JSON
        INSERT INTO OrderItems (OrderId, MenuItemId, ItemName, UnitPrice, Quantity, Note)
        SELECT @OrderId,
               JSON_VALUE(item.value, '$.menuItemId'),
               JSON_VALUE(item.value, '$.name'),
               JSON_VALUE(item.value, '$.price'),
               JSON_VALUE(item.value, '$.qty'),
               JSON_VALUE(item.value, '$.note')
        FROM OPENJSON(@ItemsJson) AS item;

        -- 3. Đẩy vào KitchenQueue
        INSERT INTO KitchenQueue (OrderId, OrderItemId, TableCode, ItemName, Quantity, Note)
        SELECT oi.OrderId, oi.OrderItemId,
               (SELECT TableCode FROM Orders WHERE OrderId = @OrderId),
               oi.ItemName, oi.Quantity, oi.Note
        FROM OrderItems oi
        WHERE oi.OrderId = @OrderId;

        -- 4. Trừ kho nguyên liệu
        INSERT INTO StockTransactions (IngredientId, TransType, Amount, Reason, OrderItemId)
        SELECT ii.IngredientId, 'out',
               ii.AmountPerServing * oi.Quantity,
               CONCAT('Order #', @OrderId, ' - ', oi.ItemName),
               oi.OrderItemId
        FROM OrderItems oi
        INNER JOIN ItemIngredients ii ON ii.MenuItemId = oi.MenuItemId
        WHERE oi.OrderId = @OrderId;

        UPDATE i
        SET i.CurrentStock = i.CurrentStock - (ii.AmountPerServing * oi.Quantity),
            i.UpdatedAt    = GETDATE()
        FROM Ingredients i
        INNER JOIN ItemIngredients ii ON ii.IngredientId = i.IngredientId
        INNER JOIN OrderItems oi      ON oi.MenuItemId   = ii.MenuItemId
        WHERE oi.OrderId = @OrderId;

        -- 5. Cập nhật DailyRevenue
        MERGE DailyRevenue AS target
        USING (SELECT CAST(GETDATE() AS DATE) AS d) AS src ON target.RevenueDate = src.d
        WHEN MATCHED THEN
            UPDATE SET TotalOrders  = TotalOrders + 1,
                       TotalAmount  = TotalAmount + @PaidAmount,
                       UpdatedAt    = GETDATE()
        WHEN NOT MATCHED THEN
            INSERT (RevenueDate, TotalOrders, TotalAmount)
            VALUES (src.d, 1, @PaidAmount);

        COMMIT;
        SELECT @OrderId AS OrderId;
    END TRY
    BEGIN CATCH
        ROLLBACK; THROW;
    END CATCH
END
GO

-- Lấy danh sách đang chờ chế biến (polling mỗi 3s)
CREATE OR ALTER PROCEDURE sp_Kitchen_GetQueue
AS
BEGIN
    SET NOCOUNT ON;
    SELECT kq.QueueId, kq.OrderId, kq.OrderItemId,
           kq.TableCode, kq.ItemName, kq.Quantity, kq.Note,
           kq.Status, kq.OrderedAt
    FROM KitchenQueue kq
    WHERE kq.Status = 'waiting'
    ORDER BY kq.OrderedAt ASC;
END
GO

-- Bếp bấm "xong" 1 món
CREATE OR ALTER PROCEDURE sp_Kitchen_ItemDone
    @QueueId    INT
AS
BEGIN
    SET NOCOUNT ON;
    UPDATE KitchenQueue
    SET Status = 'done', DoneAt = GETDATE()
    WHERE QueueId = @QueueId;

    DECLARE @OrderItemId INT;
    SELECT @OrderItemId = OrderItemId FROM KitchenQueue WHERE QueueId = @QueueId;
    UPDATE OrderItems SET ItemStatus = 'done' WHERE OrderItemId = @OrderItemId;
END
GO

-- Bếp bấm "xong tất cả bàn" → trigger in hóa đơn + thông báo tablet
CREATE OR ALTER PROCEDURE sp_Kitchen_OrderComplete
    @OrderId    INT,
    @PrinterIp  NVARCHAR(50) = NULL
AS
BEGIN
    SET NOCOUNT ON;
    BEGIN TRANSACTION;
    BEGIN TRY
        -- Đánh dấu tất cả queue items là done
        UPDATE KitchenQueue
        SET Status = 'done', DoneAt = GETDATE()
        WHERE OrderId = @OrderId AND Status = 'waiting';

        UPDATE OrderItems SET ItemStatus = 'done'
        WHERE OrderId = @OrderId AND ItemStatus = 'pending';

        -- Cập nhật Order status
        UPDATE Orders
        SET Status = 'done', CompletedAt = GETDATE()
        WHERE OrderId = @OrderId;

        -- Ghi log in (backend sẽ đọc record này để thực sự in)
        -- Kiểm tra chưa in trước
        IF NOT EXISTS (SELECT 1 FROM PrintLog WHERE OrderId = @OrderId AND Success = 1)
        BEGIN
            INSERT INTO PrintLog (OrderId, PrinterIp, Success)
            VALUES (@OrderId, @PrinterIp, 0);  -- Success=0: chờ backend in thật rồi update
        END

        COMMIT;

        -- Trả về thông tin để backend in
        SELECT o.OrderId, o.TableCode, o.TotalAmount, o.CreatedAt,
               o.CompletedAt, o.Note,
               oi.ItemName, oi.Quantity, oi.UnitPrice, oi.Note AS ItemNote
        FROM Orders o
        INNER JOIN OrderItems oi ON oi.OrderId = o.OrderId
        WHERE o.OrderId = @OrderId;
    END TRY
    BEGIN CATCH
        ROLLBACK; THROW;
    END CATCH
END
GO

CREATE OR ALTER PROCEDURE sp_Stats_Revenue
    @Period     NVARCHAR(10),   -- 'daily' | 'monthly' | 'yearly'
    @FromDate   DATE = NULL,
    @ToDate     DATE = NULL
AS
BEGIN
    SET NOCOUNT ON;
    SET @FromDate = ISNULL(@FromDate, CAST(GETDATE() AS DATE));
    SET @ToDate   = ISNULL(@ToDate,   CAST(GETDATE() AS DATE));

    IF @Period = 'daily'
        SELECT RevenueDate AS Period,
               TotalOrders, TotalAmount
        FROM DailyRevenue
        WHERE RevenueDate BETWEEN @FromDate AND @ToDate
        ORDER BY RevenueDate;

    ELSE IF @Period = 'monthly'
        SELECT FORMAT(RevenueDate, 'yyyy-MM') AS Period,
               SUM(TotalOrders) AS TotalOrders,
               SUM(TotalAmount) AS TotalAmount
        FROM DailyRevenue
        WHERE RevenueDate BETWEEN @FromDate AND @ToDate
        GROUP BY FORMAT(RevenueDate, 'yyyy-MM')
        ORDER BY Period;

    ELSE IF @Period = 'yearly'
        SELECT YEAR(RevenueDate) AS Period,
               SUM(TotalOrders) AS TotalOrders,
               SUM(TotalAmount) AS TotalAmount
        FROM DailyRevenue
        WHERE RevenueDate BETWEEN @FromDate AND @ToDate
        GROUP BY YEAR(RevenueDate)
        ORDER BY Period;
END
GO

CREATE OR ALTER PROCEDURE sp_Stats_PopularItems
    @Month  INT,
    @Year   INT,
    @TopN   INT = 10
AS
BEGIN
    SET NOCOUNT ON;
    SELECT TOP (@TopN)
           oi.MenuItemId,
           oi.ItemName,
           SUM(oi.Quantity)                         AS TotalQty,
           SUM(oi.Quantity * oi.UnitPrice)           AS TotalRevenue,
           COUNT(DISTINCT oi.OrderId)                AS TotalOrders
    FROM OrderItems oi
    INNER JOIN Orders o ON o.OrderId = oi.OrderId
    WHERE MONTH(o.CreatedAt) = @Month
      AND YEAR(o.CreatedAt)  = @Year
      AND o.Status = 'done'
    GROUP BY oi.MenuItemId, oi.ItemName
    ORDER BY TotalQty DESC;
END
GO

           CurrentStock - MinStock AS Deficit

END

-- View: trạng thái bàn hiện tại
CREATE OR ALTER VIEW vw_TableStatus AS
SELECT
    t.TableId,
    t.TableCode,
    t.TableName,
    t.Floor,
    t.Capacity,
    CASE
        WHEN ts.SessionId IS NULL THEN 'empty'
        WHEN o.OrderId IS NOT NULL THEN 'ordering'
        ELSE 'seated'
    END AS Status,
    ts.SessionId,
    ts.OpenedAt AS SessionStart,
    COUNT(DISTINCT o.OrderId)       AS TotalOrders,
    ISNULL(SUM(o.TotalAmount), 0)   AS TotalSpent
FROM Tables t
LEFT JOIN TableSessions ts  ON ts.TableId = t.TableId AND ts.Status = 'open'
LEFT JOIN Orders o          ON o.SessionId = ts.SessionId
GROUP BY t.TableId, t.TableCode, t.TableName, t.Floor, t.Capacity,
         ts.SessionId, ts.OpenedAt, o.OrderId;
GO

-- View: hàng đợi bếp đầy đủ thông tin
CREATE OR ALTER VIEW vw_KitchenQueue AS
SELECT
    kq.QueueId,
    kq.OrderId,
    kq.TableCode,
    kq.ItemName,
    kq.Quantity,
    kq.Note,
    kq.Status,
    kq.OrderedAt,
    kq.DoneAt,
    DATEDIFF(MINUTE, kq.OrderedAt, GETDATE()) AS WaitingMinutes
FROM KitchenQueue kq
WHERE kq.Status = 'waiting';
GO

-- View: doanh thu tháng này theo ngày
CREATE OR ALTER VIEW vw_RevenueThisMonth AS
SELECT
    RevenueDate,
    TotalOrders,
    TotalAmount,
    SUM(TotalAmount) OVER (
        PARTITION BY YEAR(RevenueDate), MONTH(RevenueDate)
        ORDER BY RevenueDate
        ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW
    ) AS CumulativeAmount
FROM DailyRevenue
WHERE MONTH(RevenueDate) = MONTH(GETDATE())
  AND YEAR(RevenueDate)  = YEAR(GETDATE());
GO

-- View: nguyên liệu sắp hết
CREATE OR ALTER VIEW vw_LowStockIngredients AS
SELECT
    IngredientId,
    Name,
    Unit,
    CurrentStock,
    MinStock,
    MinStock - CurrentStock AS Deficit,
    CASE
        WHEN CurrentStock = 0 THEN 'out_of_stock'
        WHEN CurrentStock <= MinStock THEN 'low'
        ELSE 'ok'
    END AS StockStatus
FROM Ingredients
WHERE CurrentStock <= MinStock;
GO

-- View: menu kèm trạng thái nguyên liệu
CREATE OR ALTER VIEW vw_MenuWithStock AS
SELECT
    mi.MenuItemId,
    mi.Name,
    mi.Price,
    mi.StockQty,
    mi.IsAvailable,
    c.Name AS CategoryName,
    -- Số lượng có thể làm theo nguyên liệu
    ISNULL(
        MIN(FLOOR(i.CurrentStock / NULLIF(ii.AmountPerServing, 0))),
        mi.StockQty
    ) AS MaxServableByIngredients
FROM MenuItems mi
INNER JOIN Categories c ON c.CategoryId = mi.CategoryId
LEFT JOIN ItemIngredients ii ON ii.MenuItemId = mi.MenuItemId
LEFT JOIN Ingredients i ON i.IngredientId = ii.IngredientId
GROUP BY mi.MenuItemId, mi.Name, mi.Price, mi.StockQty, mi.IsAvailable, c.Name;
GO
